// ✅ Your Azure Blob base URL const BASE_URL = "https://worldcuppredictions.blob.core.windows.net/worldcuppredictions/"; // ✅ List of files in your blob container const FILES = [ "cup_finals.txt", "teams.txt", "matches.txt" ]; // ✅ Load all files from Azure Blob async function loadAllFiles() { const data = {}; for (const file of FILES) { try { const response = await fetch(BASE_URL + file); if (!response.ok) { throw new Error(`Failed to load ${file}`); } const text = await response.text(); data[file] = text; } catch (error) { console.error(error); data[file] = "Error loading file."; } } return data; } // ✅ Display files on page async function displayData() { const container = document.getElementById("matches-list"); container.innerHTML = "Loading..."; const data = await loadAllFiles(); container.innerHTML = ""; for (const file in data) { const section = document.createElement("div"); section.style.marginBottom = "20px"; section.innerHTML = `

${file}

${data[file]}
`; container.appendChild(section); } } // ✅ Run on page load window.onload = displayData;
